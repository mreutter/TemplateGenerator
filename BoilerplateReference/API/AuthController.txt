    /// Constructor to enable the dependency injection of <see cref="AuthService"/>.
    /// </summary>
    /// <param name="authService">The reference to the AuthService.</param>
    public AuthController(IAuthService authService)

        _authService = authService;
    }

    [HttpPost]
    public ActionResult<LoginResponseDto?> Login(LoginRequestDto loginReq) // Custom dto
    {
        LoginResponseDto? loginRes = _authService.Verify$identifier#(loginReq); //Custom verify handling
        return loginRes == null ? Unauthorized() : Ok(loginRes);
    }
